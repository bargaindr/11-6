*/
#include <stdio.h>
#include <windows.h>
int main()
{
	int i,j,k,s;
	for(i=1;i<10;i++)
	{
		for(j=0;j<10;j++)
		{
			for(k=0;k<10;k++)
			{
				s=i*100+j*10+k;
				printf("%d\n",s);
				Sleep(1000);

			}
		}
	}
	return 0;
}
